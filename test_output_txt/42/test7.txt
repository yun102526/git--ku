// read读数→[]=存入数组→=[]读取数组→+累加求和
// 输入: 3个整数, 如 10, 20, 30
// 预期: 10+20+30=60

3: (read, _, _, t1)
4: ([]=, t1, i, arr)
5: (=[], arr, i, t2)
6: (+, sum, t2, t3)
7: (=, t3, _, sum)
8: (+, i, 1, t4)
9: (=, t4, _, i)
10: (J, _, _, 2)
11: (write, sum, _, _)
12: (return, sum, _, _)

--- [4.2 输出] LLVM IR 代码 ---
declare i32 @printf(ptr, ...)

@.fmt_num = private constant [3 x i8] c"%d\00"
@.fmt_chr = private constant [3 x i8] c"%c\00"

@input_data = global [3 x i32] [i32 10, i32 20, i32 30]
@input_idx = global i32 0

define i32 @main() {
entry:
  %arr_ptr = alloca [100 x i32]
  %i_ptr = alloca i32
  %sum_ptr = alloca i32
  br label %block0

block0:
  store i32 0, ptr %i_ptr
  store i32 0, ptr %sum_ptr
  br label %block1

block1:
  %rt0 = load i32, ptr %i_ptr
  %brc_2 = icmp sge i32 %rt0, 3
  br i1 %brc_2, label %block3, label %block2

block2:
  %ld_3 = load i32, ptr @input_idx
  %gp_3 = getelementptr [3 x i32], ptr @input_data, i32 0, i32 %ld_3
  %t1 = load i32, ptr %gp_3
  %ni_3 = add i32 %ld_3, 1
  store i32 %ni_3, ptr @input_idx
  %rt1 = load i32, ptr %i_ptr
  %gp_4 = getelementptr [100 x i32], ptr %arr_ptr, i32 0, i32 %rt1
  store i32 %t1, ptr %gp_4
  %rt2 = load i32, ptr %i_ptr
  %gp_t2 = getelementptr [100 x i32], ptr %arr_ptr, i32 0, i32 %rt2
  %t2 = load i32, ptr %gp_t2
  %rt3 = load i32, ptr %sum_ptr
  %t3 = add i32 %rt3, %t2
  store i32 %t3, ptr %sum_ptr
  %rt4 = load i32, ptr %i_ptr
  %t4 = add i32 %rt4, 1
  store i32 %t4, ptr %i_ptr
  br label %block1

block3:
  %rt5 = load i32, ptr %sum_ptr
  call i32 (ptr, ...) @printf(ptr @.fmt_num, i32 %rt5)
  %rt6 = load i32, ptr %sum_ptr
  ret i32 %rt6

}
