--- [4.2 输出] LLVM IR 代码 ---
declare i32 @printf(ptr, ...)

@.fmt_num = private constant [3 x i8] c"%d\00"
@.fmt_chr = private constant [3 x i8] c"%c\00"


define i32 @main() {
entry:
  %a_ptr = alloca i32
  %b_ptr = alloca i32
  %r_ptr = alloca i32
  br label %block0

block0:
  store i32 17, ptr %a_ptr
  store i32 5, ptr %b_ptr
  %rt0 = load i32, ptr %a_ptr
  %rt1 = load i32, ptr %b_ptr
  %t1 = srem i32 %rt0, %rt1
  store i32 %t1, ptr %r_ptr
  %rt2 = load i32, ptr %r_ptr
  call i32 (ptr, ...) @printf(ptr @.fmt_num, i32 %rt2)
  %rt3 = load i32, ptr %r_ptr
  ret i32 %rt3

}
