// i从10递减到1, sum累加i, 结果=10+9+...+1=55
// 测试四元式: do-while先执行再判断, 使用J>条件跳转
0: (=, 10, _, i)
1: (=, 0, _, sum)
2: (+, sum, i, t1)
3: (=, t1, _, sum)
4: (-, i, 1, t2)
5: (=, t2, _, i)
6: (J>, i, 0, 2)
7: (write, sum, _, _)
8: (return, sum, _, _)

--- [4.2 输出] LLVM IR 代码 ---
declare i32 @printf(ptr, ...)

@.fmt_num = private constant [3 x i8] c"%d\00"
@.fmt_chr = private constant [3 x i8] c"%c\00"


define i32 @main() {
entry:
  %i_ptr = alloca i32
  %sum_ptr = alloca i32
  br label %block0

block0:
  store i32 10, ptr %i_ptr
  store i32 0, ptr %sum_ptr
  br label %block1

block1:
  %rt0 = load i32, ptr %sum_ptr
  %rt1 = load i32, ptr %i_ptr
  %t1 = add i32 %rt0, %rt1
  store i32 %t1, ptr %sum_ptr
  %rt2 = load i32, ptr %i_ptr
  %t2 = sub i32 %rt2, 1
  store i32 %t2, ptr %i_ptr
  %rt3 = load i32, ptr %i_ptr
  %brc_6 = icmp sgt i32 %rt3, 0
  br i1 %brc_6, label %block1, label %block2

block2:
  %rt4 = load i32, ptr %sum_ptr
  call i32 (ptr, ...) @printf(ptr @.fmt_num, i32 %rt4)
  %rt5 = load i32, ptr %sum_ptr
  ret i32 %rt5

}
