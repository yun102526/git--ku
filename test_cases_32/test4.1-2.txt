=== 3.2 递归：单递归 求1~n之和 ===
// 测试递归函数调用 - seq(n) = n + seq(n-1)
// 计算1到n的和
// 输入: n, 范围1~100
int seq(int);
main()
{
   int n;
   n=read();
   write(seq(n));
}

int seq(int m) {
  int i,fa;
  i=0;
  if(m<=1)
    {
	fa=1;
     }
   else{
         fa=m+seq(m-1);
     }
   return fa;
}
=== 输入: 10 (1~100的整数) ===
=== 预期输出: 55 ===
