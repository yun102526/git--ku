=== 3.2 嵌套控制：while+if 求最大最小平均值 ===
// 测试while与if嵌套, 输入5个整数求最大值、最小值、平均值
// 输入: 5个整数, 如 10, 5, 8, 3, 9
// 输出: 最大值、最小值、(和/5的整数商)
void main()
{
	int a=0;
	int max = 0;
	int min = 0;
	int sum = 0;
	int i = 0;
	while(i<5)
	{
		a = read();
		if(i==0){
			max = a;
			min = a;}
		else
		{
			if(a>max){max=a;}
			if(a<min){min=a;}
		}
		sum = sum+a;
		i = i+1;
	}
	write(max);
	write(min);
	sum = sum/5;
	write(sum);
}
=== 输入: 10, 5, 8, 3, 9 ===
=== 预期输出: 10 3 7 ===
