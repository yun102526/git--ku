=== 3.2 递归：单递归 求阶乘 ===
// 测试递归函数调用 - factor(n) = n!
// 使用int类型, 结果需小于65536
// 输入: n, 范围0~8
// 注意: 不同函数使用不同参数名m和n, 需验证变量作用域
int factor(int);
main()
{
  int n;
  n=read();
  write(factor(n));
}

int factor(int m) {
  int fa;
  if(m<=1) {
    fa=1;
  }
  else{
    fa=m*factor(m-1);
  }
  return fa;
}
=== 输入: 5 ===
=== 预期输出: 120 ===
