int main(void) {
    int n, i, sum, flag, arr[5], result;
    n = read();
    i = 0;
    sum = 0;
    flag = 1;
    while (i < 5) {
        arr[i] = read();
        i = i + 1;
    }
    for (i = 0; i < 5; i = i + 1) {
        if (arr[i] < 0)
            continue;
        if (arr[i] == 0)
            break;
        sum = sum + arr[i];
    }
    do {
        sum = sum + 1;
        n = n - 1;
    } while (n > 0);
    if ((sum > 10 && flag == 1) || n == 0) {
        result = addAdjust(sum, 2);
    } else {
        result = sum / 2 + sum % 2;
    }
    write(result);
    return result;
}
int addAdjust(int x, int y) {
    int r;
    r = x + y;
    if (r > 20) {
        r = r - 3;
    } else {
        r = r + 3;
    }
    return r;
}
