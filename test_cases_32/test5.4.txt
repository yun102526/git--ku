=== 3.2 数组：斐波那契数列(数组存储) ===
// 测试数组声明、赋值和读取
// 输入n(范围1~49), 计算第1~n个斐波那契数并输出
// a[0]=0, a[1]=1, a[i]=a[i-1]+a[i-2]
int a[50];
int main(void)
{
    int i;
    int n;
    n = read();
    a[0] = 0;
    a[1] = 1;

    i = 2;
    while(i <= n)
    {
        a[i] = a[i-1] + a[i-2];
        i = i + 1;
    }
    i = 1;
    while(i <= n)
    {
      write(a[i]);
        i = i + 1;
    }
    return 0;
}
=== 输入: 8 (1~49的整数) ===
=== 预期输出: 1 1 2 3 5 8 13 21 ===
