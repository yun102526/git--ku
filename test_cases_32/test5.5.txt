=== 3.2 数组：冒泡排序 ===
// 测试数组读写、双重for循环、if条件交换
// 输入5个整数(范围不限), 对数组进行冒泡排序(升序)
// 每轮排序后打印当前数组状态
// 输入: 5个整数, 如 5, 3, 4, 1, 2
main() 
{
	int a[5];
	int b=5;
	int i = 0;
	int j=0;
	int temp;
	int k;
	while(i<b)
	{
		a[i]= read();
		i = i+1;
	}
	for(i=0;i<5;i++) {  //打印原始数据
	    write(a[i]);
	    write(' ');
	}
	write('\n');
	for(i=3;i>=0;i=i-1)
	{
		for(j=0;j<=i;j=j+1)
		{	
			b = j+1;
			if(a[j]>a[b])
			{
				temp=a[j];
				a[j]=a[b];
				a[b]=temp;
			}
		}
		for(k=0;k<5;k=k+1) {  //打印交换后的数据
	   		 write(a[k]);
			 write(' ');
		}
		write('\n');
	}
}
=== 输入: 5, 3, 4, 1, 2 (5个整数) ===
=== 预期输出: 多行, 每行5个升序排列的整数 ===
