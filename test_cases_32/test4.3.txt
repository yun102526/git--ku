=== 3.2 递归：三递归 上楼梯问题 ===
// 测试三递归函数调用 - f(n)=f(n-1)+f(n-2)+f(n-3)
// 一次可上1/2/3级台阶, 求上n级台阶的方法数
// 结果序列: 1 2 4 7 13 24 44 81 149...
// 输入: n, 范围1~10(太大会很慢)
int f(int);
main()
{
   int m;
   m = read();
   write(f(m));

}

int f(int n) {
   int a,b,c;
   if(n==1) {
		return 1;
	}
   else if(n==2) {
		return 2;
	}
   else if(n==3) {
		return 4;
	}
    a = f(n - 1);
    b = f(n - 2);
    c = f(n - 3);
    return a + b + c;
 }
=== 输入: 6 (1~10的整数) ===
=== 预期输出: 24 ===
