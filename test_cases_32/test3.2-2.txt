=== 3.2 函数：判断素数 ===
// 测试函数调用、if条件、while循环的综合运用
// 输入一个正整数num(1~100), 判断是否为素数
// isPrime返回1(是素数)或0(不是)
// 输入: 正整数, 范围1~100
int isPrime(int x);

int main(void)
{
    int num;
    int res;
    read(num);
    if(isPrime(num) == 1)
    {
        res = 1;
    }
    else
    {
        res = 0;
    }
    write(res);
    return 0;
}
int isPrime(int x)
{
    int i;
    int yes;
    i = 2;
    if(x < 2)
    {
        yes = 0;
    }
    else
    {
        yes = 1;
    }
    while(i * i <= x)
    {
        if(x / i * i == x)
        {
            yes = 0;
        }
        i = i + 1;
    }
    return yes;
}
=== 输入: 7 ===
=== 预期输出: 1 ===
