=== 3.2 控制流：for+break 求素数 ===
// 测试for循环嵌套中的break语句
// 输入一个正整数n(2~100), 输出2~n之间所有素数
// break用于跳出内层循环
void main()
{
	int n=0;
	int i=0;
	int j=0;
	n = read();
	for(i=2;i<n;i=i+1)
	{
		for(j=2;j<i;j=j+1)
		{
			if(i%j==0)
			{break;}
		}
		if(j==i)
		{write(i);}
	}
}
=== 输入: 20 ===
=== 预期输出: 2 3 5 7 11 13 17 19 ===
