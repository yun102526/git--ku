=== 3.2 嵌套控制：双重for 求N以内素数 ===
// 测试双重for循环和整除判断
// 输入一个正整数N(2~100), 输出2~N之间的所有素数
main(){

    int N = read() ;
    int count=0,nprime=0,i,j;
    for(i=2;i<=N;i=i+1) {
       nprime = 0;
       for(j=2;j<i;j=j+1) {
	   if(i%j == 0) nprime = nprime + 1;
       }
       if(nprime == 0) {
            write(i);
            count = count + 1;
        }
     }

}
=== 输入: 20 (2~100的整数) ===
=== 预期输出: 2 3 5 7 11 13 17 19 ===
