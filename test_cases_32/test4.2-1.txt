=== 3.2 递归：双递归 求斐波那契数列 ===
// 测试双递归函数调用 - seq(m)=seq(m-1)+seq(m-2)
// 计算斐波那契数列第n项
// 输入: n, 范围1~15 (太大递归会很慢)
// 结果序列: 1 1 2 3 5 8 13 21 34...
int seq(int);
main()
{
   int n;
   n=read();
   write(seq(n));
}

int seq(int m) 
{
      int s,a,b;
      if(m<=2)
          s=1;
      else{
           a=seq(m-1);
           b=seq(m-2);
           s=a+b;
      }
      return s;

}
=== 输入: 8 (1~15的整数) ===
=== 预期输出: 21 ===
