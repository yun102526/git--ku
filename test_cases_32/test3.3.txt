=== 3.2 函数：函数嵌套调用 sum(sum(3,4), N) ===
// 测试函数的嵌套调用(一个函数调用另一个函数的结果)
// sum(x,y)=x+y, 结果=sum(sum(3,4), N)=7+N
// 输入: 一个整数N(范围不限)
int a = 1 ;

int sum(int,int);

main(){

    int N = read() ;

    a = sum(sum(3,4),N) ;
    write(a);

}

int sum(int sum_x,int sum_y){

    int result ;
    result = sum_x + sum_y ;

    return result ;

}
=== 输入: 10 ===
=== 预期输出: 17 ===
