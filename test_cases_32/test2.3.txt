=== 3.2 嵌套控制：while+if 求1~N内偶数和 ===
// 测试while循环嵌套if条件
// 输入一个正整数N, 求1到N-1内所有偶数的和
// 输入: 1~100的整数
int a = 1 ;

main(){

    int i = 1,sum=0,N ;
    N = read();

    while(i < N){
       if(i%2 == 0){
           sum = sum +i;
       }
       i = i + 1;
    }
    write(sum);
}
=== 输入: 10 ===
=== 预期输出: 20 ===
