=== 3.2 嵌套控制：双重for循环 打印三角形 ===
// 测试多重for循环和字符输出(write)
// 输入行数n(1~10), 打印n行直角三角形图案
// 每行先打印空格, 再打印星号
void main()
{
	int n=0,j=0,i=0;
	n = read();
	for(i=0;i<n;i=i+1)
	{
		for(j=0;j<n-i-1;j=j+1)
		{
			write(' ');	
		}
		for(j=0;j<i+1;j=j+1)
		{
			write('*');
			write(' ');
		}
		write('\n');
	}
}
=== 输入: 5 ===
=== 预期输出: 打印三角形图案 ===
