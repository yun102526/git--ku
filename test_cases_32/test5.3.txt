// 与test4.3不同: a=f(n-1)+f(n-2)+f(n-3)直接用表达式
// 输入: n, 范围1~10
int f(int);
main()
{
   int m;
   m = read();
   write(f(m));

}

int f(int n) {
   int a;
   if(n==1) {
		return 1;
	}
   else if(n==2) {
		return 2;
	}
   else if(n==3) {
		return 4;
	}
    a = f(n - 1)+ f(n - 2)+ f(n - 3);
    return a ;
 }
