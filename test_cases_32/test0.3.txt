=== 3.2 基本表达式：关系运算 > < == != ===
// 测试大于、小于、等于、不等于关系运算符
// a=1>-2=1, b=1<-2=0, c=1==-2=0, d=1!=-2=1
main() {
       int x = 1;
       int y = -2;
       int a, b, c, d;
       a = x > y;
       b = x < y;
       c = x == y;
       d = x != y;
       write(a);
       write(b);
       write(c);
       write(d);
       }

=== 预期输出: 1 0 0 1 ===
