=== 3.2 嵌套控制：if+while 求阶乘 ===
// 测试if与while的组合使用
// 输入一个非负整数n, 计算n的阶乘(0的阶乘为0)
// 输入范围: 0~8 (结果不超过65535)
main()
{
  int i,factor,n;
  i=0;
  n=read();
  if(n<1)     {
      factor=0;
    }
  else {
    factor=1;
  }
  while(i<n)
    {
      i=i+1;
      factor=factor*i;
    }
  write(factor);
}
=== 输入: 5 ===
=== 预期输出: 120 ===
