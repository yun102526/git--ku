=== 3.2 递归：单递归 最大公约数(gcd) ===
// 测试递归函数 - gcd(x,y)求最大公约数(欧几里得算法)
// 输入两个正整数a和b(1~1000)
int gcd(int, int);

int main(void)
{
    int a;
    int b;
    int c;
    a = read();
    b = read();
    c = gcd(a, b);
    write(c);
    return 0;
}
int gcd(int x, int y)
{
    if(y == 0)
    {
        return x;
    }
    else
    {
        return gcd(y, x-x/y*y);
    }
    return 0;
}
=== 输入: 48, 18 (两个正整数, 1~1000) ===
=== 预期输出: 6 ===
