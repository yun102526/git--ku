=== 3.2 嵌套控制：双重for 打印乘法表 ===
// 测试函数调用+双重for循环
// 输入一个正整数n(1~9), 输出n×n乘法表
// 注意: ""字符串会转为整数值求和, 输出为数字
int chengfabiao(int n)
{
  int i,j,s;
  for(i=1;i<=n;i=i+1)
        {
         for(j=1;j<=i;j=j+1)
        {
             s=i*j;
           write(j);write(i);write(s);
       }
    }
  return 0;
}




main()
{
   int n;
   n=read();
   chengfabiao(n);
}
=== 输入: 4 ===
=== 预期输出: 乘法表各乘积 ===
