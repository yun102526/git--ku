=== 3.2 递归：双递归 组合数 C(n,k) ===
// 测试双递归函数调用 - 计算组合数 C(m,k)
// comp(n,i)=comp(n-1,i)+comp(n-1,i-1); 边界: n==i或i==0返回1
// 输入: m和k, m范围1~15, k范围0~m
int comp(int,int);
main()
{
   int m,k,result;
   m = read();
   k = read();
   result = comp(m,k);
   write(result);
}

int comp(int n,int i) {
int a, b; 
if(n == i || i == 0)
	{
		return 1;
	}
	 a = comp(n-1,i) ;
                 b = comp(n-1,i-1);
	return a + b ; 
}
=== 输入: 5 2 ===
=== 预期输出: 10 ===
