=== 3.2 递归：双递归 组合数(紧凑写法) ===
// 测试双递归函数调用(嵌套表达式写法)
// 与test4.2-2不同: a=comp(n-1,i)+comp(n-1,i-1)直接用表达式
// 输入: m和k, m范围1~15, k范围0~m
int comp(int,int);
main()
{
   int m,k,result;
   m = read();
   k = read();
   result = comp(m,k);
   write(result);
}

int comp(int n,int i) {
int a; 
if(n == i || i == 0)
	{
		return 1;
	}
	 a = comp(n-1,i) + comp(n-1,i-1);
	return a ; 
}
=== 输入: 6, 3 (m=6, k=3, 1~15) ===
=== 预期输出: 20 ===
