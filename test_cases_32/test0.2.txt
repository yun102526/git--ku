=== 3.2 基本表达式：逻辑运算 && 和 || ===
// 测试逻辑与(&&)和逻辑或(||)运算符
// a=2&&3=1, b=2&&0=0, c=2||3=1, d=2||0=1
main() {
       int x = 2;
       int y = 3;
       int a, b, c, d;
       a = x && y;
       b = x && 0;
       c = x || y;
       d = x || 0;
       write(a);
       write(b);
       write(c);
       write(d);
       }	
=== 输入: (无) ===
=== 预期输出: 1 0 1 1 ===
