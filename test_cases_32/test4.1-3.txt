=== 3.2 递归：单递归 反序输出整数各位 ===
// 测试递归函数 - invert(n)逐位反序输出
// 输入一个正整数(小于65535), 逐个输出其各位数字(个位在前)
// 输入: 1~65534的整数
void invert(int);
main()
{
   int k;
   k=read();
   invert(k);
}

void invert(int m) {
  int n;
  if(m>0&&m<10)
    {

	write(m);
     }
   else{
         n=m%10;
         write(n);
         n=m/10;
         invert(n);
     }
   return 0;
}
=== 输入: 1234 (1~65534的整数) ===
=== 预期输出: 4 3 2 1 ===
